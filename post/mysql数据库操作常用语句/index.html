MySQL数据库操作常用语句

数据库常用语句

数据库

-- 需要自己输入的为中文，“[ ]”内为选填项

-- 创建数据库
create database 数据库 [character set 字符集] [collate 校对规则];
-- 查看所有数据库
show database;
-- 查看数据库的定义结构
show create database 数据库;
-- 删除数据库
drop database 数据库;
-- 修改数据库：只能修改字符集，utf-8写作utf8
alter database 数据库 character set 字符集;
-- 切换数据库
use 数据库;
-- 查看正在使用的数据库
select database();

数据表

-- 创建数据表--------------------------------------------------
create table 表名(
	字段 字段类型 [约束],
	......
	字段 字段类型 [约束]
);
-- 约束
not null	-- 不为空
unique 		-- 唯一
primary key	-- 主键,一般用于id,后面一般加 auto_increment //自增

-- 查看表--------------------------------------------------
-- 查看所有表
show tables;
-- 查看表的定义结构
desc 表;

-- 修改表--------------------------------------------------
-- 增加字段
alter table 表 add 字段 类型 约束;
-- 修改列的类型约束
alter table 表 字段 类型 约束;
-- 修改字段的名称、类型、约束
alter table 表 change 旧字段 新字段 类型 约束;
-- 删除字段
alter table 表 drop 字段;
-- 修改表名
rename table 旧表 to 新表;

-- 删除表--------------------------------------------------
drop table 表;

记录

增删改

-- 增--------------------------------------------------
-- 插入特定字段
insert into 表(列,列,...) values(值,值,...)
-- 插入所有字段
insert into 表 values(值,值,...);		-- 所有值

-- 改--------------------------------------------------
update 表 set 字段=值,... [where 条件];

-- 删--------------------------------------------------
-- 删除记录
delete from 表 [where 条件];
-- 清空记录
truncate table 表;	-- 相当于删除表,再创建表

单表查询

-- 查--------------------------------------------------
-- 单表查询-----
-- 基本语法
select [*][字段,字段,...][字段 as 别名,...] [distinct 字段] from 表 [where 条件]
-- 查询所有记录
select * from 表;
-- 查询特定记录
select 字段,字段,... from 表;
-- 去重查询
select distinct 字段 from 表;
-- 别名查询
select 字段 [as] 别名,字段 ... from 表;
select 别名.* from 表 as 别名;
-- 条件查询
select ... from 表 where 条件;
-- 比较运算符
> < >= <= = <>|!=	-- 大于 小于 大于等于 小于等于 不等于
between...and...	-- 显示再某一区间的值
in(set)				-- 显示再in字段中的值,例:in(1,2,3)
like 'set'			-- 模糊查询,一般和"_"或者"%"一起使用;"_"占一位,"%"占任意位
is null				-- 判断不为空
-- 逻辑运算符,可以用符号表示:& | !
and					-- 多个条件同时成立
or					-- 多个条件任一成立
not					-- 不成立

排序查询

-- 排序查询--------------------------------------------------
-- 单列排序
select ... from 表 [where 条件] order by 字段 [asc|desc];	--升序|降序
-- 组合排序
select ... from 表 [where 条件] order by 字段 [asc|desc], 字段名 [asc|desc], ...;

-- 聚合函数
select 聚合函数(字段) from 表 [where 条件];
max(字段)		-- 求该字段的最大值
min(字段)		-- 求该字段的最小值
avg(字段)		-- 求该字段的平均值
sum(字段)		-- 求该字段的总和
count(字段)	-- 统计该字段的记录数量
-- 聚合函数会忽略空值
-- 不忽略空值
select 聚合函数(ifnull(字段,0)) from 表 [where 条件];

-- 分组查询
select 字段1,字段2,... from 表 [where 条件] group by 列 [having 条件];

-- 分页查询
select ... from 表 [where 条件] limit 起始行数,查询的记录条数;

多表查询

-- 多表查询--------------------------------------------------
-- 添加外键----------------
-- 1.新建表时增加外键
create table 表(
	字段 字段类型 约束;
    ......
    字段 字段类型 约束,
    [constraint] [外键约束名称] foreign key(外键字段) references 主表(主键字段)
);
-- 关键字解释
constraint		-- 约束关键字
foreign key 外键字段		-- 某个字段作为外键
references 主表(主键字段)		-- 参照表中的某个字段
-- 2.已有表增加外键
alter table 从表 add [constraint] [外键约束名称] foreign key(外键字段) references 主表(主键字段);
-- 删除外键--
alter table 表 deop foreign key 外键名称;
-- 外键级联操作--
-- 在增加外键的后面续写
on update cascade	-- 关联更新
on delete cascade	-- 关联删除

连接查询

-- 连接查询--------------------------------------------------
-- 交叉查询-----
select ... from 表1,表2,...;
select a.字段, b.字段 from a,b;
select a.*, b.* from a,b;
select * from a,b;

-- 内连接查询--------------
-- 隐式内连接
select ... from 表1, 表2,... where 连接条件;
-- 显示内连接
select ... from a [inner] join b on 连接条件 [where 其他条件];

-- 外连接查询--------------
-- 左外连接-左边的表为主表
select [字段][*] from 表1 left [outer] join 表2 on 条件;
-- 右外连接-右边的表为主表
select [字段][*] from 表1 right [outer] join 表2 on 条件;

子查询

-- 子查询-----------------
-- 子查询结果为单个
select ... from 表 where 字段 [比较条件] (select 字段 from 表 [where 条件]);
-- 子查询结果为单列多行
select ... from 表 where 字段 in(select 字段 from 表 [where 条件]);
-- 子查询结果为多行多个
select ... from (select * from 表 [where 条件]) as 别名 [where 条件];

事务

-- 事务-----------------------------------------------------
-- 自动事务(MySQL默认):一条sql语句就是一个事务
-- 手动开启一个事务
-- 1.
start transaction;	-- 开启事务
commit;				-- 提交事务
rollback;			-- 回滚事务
-- 2.
-- 查看MySQL中事务是否自动提交
show variables like '%commit%';
-- 设置自动提交事务的参数为OFF
set autocommit = 0;	-- 0:OFF 1:ON
-- 回滚点
-- 设置回滚点
savepoint 标记;
-- 回到回滚点
rollback to 标记;

-- 数据库备份和还原-------------------------------------------
-- cmd备份
mysqldump -u用户名 -p密码 数据库 > 文件路径
-- 还原
source 导入文件
